pipeline {
    agent any
    stages {
        stage('git repo & clean') {
            steps {
               
                bat "git clone https://github.com/achyut009/Bookevents.git"
                bat "mvn clean -f Bookevents"
            }
        }
        stage('install') {
            steps {
                bat "mvn install -f Bookevents"
            }
        }
        stage('test') {
            steps {
                bat "mvn test -f Bookevents"
            }
        }
        stage('package') {
            steps {
                bat "mvn package -f Bookevents"
            }
        }
    }
}
